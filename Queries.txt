# Konstruiere adressen.ttl 

PREFIX leoplace: <http://le-online.de/ontology/place/ns#> 
construct { 
  ?a a leoplace:Place; leoplace:address ?b; 
  leoplace:fixedAddress ?b; 
  leoplace:placeName ?n; leoplace:fixedPlaceName ?n . 
}
where {
?a leoplace:address ?b; leoplace:placeName ?n .
}

# Lade ld-adressen.ttl als http://le-online.de/places/ in den RDF-Store unter
# leipzig-data.de/Data. 

# Frage den SPARQL-Endpunkt http://leipzig-data.de:8890/sparql 

PREFIX leoplace: <http://le-online.de/ontology/place/ns#> 
PREFIX ld: <http://leipzig-data.de/Data/Model/> 
PREFIX geosparql: <http://www.opengis.net/ont/geosparql#> 
select distinct ?b xsd:string(?g)
from <http://le-online.de/places/>
from <http://leipzig-data.de/Data/Adressen/>
from <http://leipzig-data.de/Data/GeoDaten/>
where {
?a leoplace:ldAddress ?b .
?b a ld:Adresse; geosparql:asWKT ?g . 
}
